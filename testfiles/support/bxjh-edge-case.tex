%% \bxjh_test_print_result_with_date:Nnnnn \CS
%%   { <year> } { <month> } { <day> } { <code> }
\cs_new:Npn \bxjh_test_print_result_with_date:Nnnnn #1#2#3#4#5
  {
    \tl_set:Nf \l_tmpa_tl { #1 { #2 } { #3 } { #4 } #5 }

    \TYPE
      {
        \int_eval:n { #2 } / \int_eval:n { #3 } / \int_eval:n { #4 } ~
        [ \exp_not:o { \l_tmpa_tl } ]
      }
  }

%% \TestHolidayName { <year> } { <month> } { <day> }
